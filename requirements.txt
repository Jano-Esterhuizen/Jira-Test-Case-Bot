"""
Generate QA test cases from a Jira ticket + its linked GitHub PR,
then post them back to the ticket as a comment.

Triggered by GitHub Actions via repository_dispatch.

Required env vars:
  ISSUE_KEY          e.g. "PROJ-123" (sent by the Jira Automation webhook)
  JIRA_BASE_URL      e.g. "https://yourcompany.atlassian.net"
  JIRA_EMAIL         Jira account email (for API auth)
  JIRA_API_TOKEN     Jira API token (https://id.atlassian.com/manage-profile/security/api-tokens)
  ANTHROPIC_API_KEY  Claude API key (https://console.anthropic.com)
  GITHUB_TOKEN       Provided automatically by GitHub Actions
  GITHUB_REPOSITORY  e.g. "yourorg/yourrepo" (provided automatically)
"""

import os
import sys
import requests
from anthropic import Anthropic

# ---------- Config ----------
MODEL = "claude-sonnet-4-6"
MAX_DIFF_CHARS = 60_000  # keep prompt size sane on very large PRs
MAX_OUTPUT_TOKENS = 4_000

ISSUE_KEY = os.environ["ISSUE_KEY"].strip()
JIRA_BASE_URL = os.environ["JIRA_BASE_URL"].rstrip("/")
JIRA_AUTH = (os.environ["JIRA_EMAIL"], os.environ["JIRA_API_TOKEN"])
GH_TOKEN = os.environ.get("GH_PAT") or os.environ["GITHUB_TOKEN"]
GH_REPO = os.environ["GITHUB_REPOSITORY"]

GH_HEADERS = {
    "Authorization": f"Bearer {GH_TOKEN}",
    "Accept": "application/vnd.github+json",
    "X-GitHub-Api-Version": "2022-11-28",
}


# ---------- Jira ----------
def adf_to_text(node) -> str:
    """Flatten Jira's Atlassian Document Format into plain text."""
    if node is None:
        return ""
    if isinstance(node, str):
        return node
    out = []
    if isinstance(node, dict):
        if node.get("type") == "text":
            out.append(node.get("text", ""))
        for child in node.get("content", []) or []:
            out.append(adf_to_text(child))
        if node.get("type") in ("paragraph", "heading", "listItem"):
            out.append("\n")
    elif isinstance(node, list):
        for child in node:
            out.append(adf_to_text(child))
    return "".join(out)


def fetch_jira_issue(key: str) -> dict:
    url = f"{JIRA_BASE_URL}/rest/api/3/issue/{key}"
    resp = requests.get(url, auth=JIRA_AUTH, timeout=30)
    resp.raise_for_status()
    fields = resp.json()["fields"]

    description = adf_to_text(fields.get("description")).strip()

    # Acceptance criteria often lives in a custom field; collect any custom
    # field whose name mentions "acceptance" so we don't miss it.
    ac_parts = []
    names_url = f"{JIRA_BASE_URL}/rest/api/3/issue/{key}?expand=names"
    names_resp = requests.get(names_url, auth=JIRA_AUTH, timeout=30)
    if names_resp.ok:
        data = names_resp.json()
        for field_id, field_name in (data.get("names") or {}).items():
            if "acceptance" in (field_name or "").lower():
                value = data["fields"].get(field_id)
                text = adf_to_text(value).strip() if value else ""
                if text:
                    ac_parts.append(text)

    return {
        "key": key,
        "summary": fields.get("summary", ""),
        "issue_type": (fields.get("issuetype") or {}).get("name", ""),
        "description": description,
        "acceptance_criteria": "\n".join(ac_parts),
    }


def post_jira_comment(key: str, body_text: str) -> None:
    """Post a plain-text comment using ADF."""
    url = f"{JIRA_BASE_URL}/rest/api/3/issue/{key}/comment"
    payload = {
        "body": {
            "type": "doc",
            "version": 1,
            "content": [
                {
                    "type": "paragraph",
                    "content": [{"type": "text", "text": body_text}],
                }
            ],
        }
    }
    resp = requests.post(url, json=payload, auth=JIRA_AUTH, timeout=30)
    resp.raise_for_status()


# ---------- GitHub ----------
import re

PR_URL_PATTERN = re.compile(
    r"https://github\.com/([\w.\-]+)/([\w.\-]+)/pull/(\d+)", re.IGNORECASE
)


def find_pr_url_in_comments(issue_key: str) -> tuple[str, str, int] | None:
    """Scan the Jira ticket's comments (and description) for a GitHub PR URL.

    Returns (owner, repo, pr_number) from the most recent match, or None.
    """
    url = f"{JIRA_BASE_URL}/rest/api/3/issue/{issue_key}/comment"
    resp = requests.get(url, auth=JIRA_AUTH, params={"orderBy": "-created"}, timeout=30)
    texts = []
    if resp.ok:
        for comment in resp.json().get("comments", []):
            texts.append(adf_to_text(comment.get("body")))

    # Also check the description itself, in case the link lives there
    desc_resp = requests.get(
        f"{JIRA_BASE_URL}/rest/api/3/issue/{issue_key}", auth=JIRA_AUTH, timeout=30
    )
    if desc_resp.ok:
        texts.append(adf_to_text(desc_resp.json()["fields"].get("description")))

    for text in texts:  # newest comment first
        match = PR_URL_PATTERN.search(text or "")
        if match:
            owner, repo, number = match.group(1), match.group(2), int(match.group(3))
            return owner, repo, number
    return None


def fetch_pr(owner: str, repo: str, pr_number: int) -> dict | None:
    url = f"https://api.github.com/repos/{owner}/{repo}/pulls/{pr_number}"
    resp = requests.get(url, headers=GH_HEADERS, timeout=30)
    if not resp.ok:
        print(f"Warning: could not fetch PR {owner}/{repo}#{pr_number} "
              f"({resp.status_code}). Does the token have access to that repo?")
        return None
    return resp.json()


def find_linked_pr_by_key(issue_key: str) -> dict | None:
    """Fallback: find the most recent PR whose branch name or title contains the issue key."""
    url = f"https://api.github.com/search/issues"
    query = f"repo:{GH_REPO} is:pr {issue_key} in:title"
    resp = requests.get(url, headers=GH_HEADERS, params={"q": query, "sort": "updated"}, timeout=30)
    items = resp.json().get("items", []) if resp.ok else []

    if not items:
        # Fall back to scanning open + recent PRs for the key in the branch name
        prs_url = f"https://api.github.com/repos/{GH_REPO}/pulls"
        resp = requests.get(
            prs_url, headers=GH_HEADERS,
            params={"state": "all", "sort": "updated", "direction": "desc", "per_page": 50},
            timeout=30,
        )
        resp.raise_for_status()
        for pr in resp.json():
            branch = pr.get("head", {}).get("ref", "")
            if issue_key.lower() in branch.lower() or issue_key.lower() in pr.get("title", "").lower():
                return pr
        return None

    pr_number = items[0]["number"]
    pr_url = f"https://api.github.com/repos/{GH_REPO}/pulls/{pr_number}"
    resp = requests.get(pr_url, headers=GH_HEADERS, timeout=30)
    resp.raise_for_status()
    return resp.json()


def fetch_pr_diff(owner: str, repo: str, pr_number: int) -> str:
    url = f"https://api.github.com/repos/{owner}/{repo}/pulls/{pr_number}"
    headers = {**GH_HEADERS, "Accept": "application/vnd.github.v3.diff"}
    resp = requests.get(url, headers=headers, timeout=60)
    resp.raise_for_status()
    diff = resp.text
    if len(diff) > MAX_DIFF_CHARS:
        diff = diff[:MAX_DIFF_CHARS] + "\n\n[...diff truncated due to size...]"
    return diff


# ---------- Claude ----------
SYSTEM_PROMPT = """You are a senior QA / test automation engineer. You write precise,
practical test cases that another tester can execute directly.

Given a Jira ticket (description + acceptance criteria) and the code diff of the
linked pull request, produce a test plan with these sections:

1. SUMMARY OF CHANGE — 2-3 sentences on what changed and the risk areas, based on the diff.
2. HAPPY PATH TEST CASES
3. EDGE CASES & BOUNDARY CONDITIONS
4. NEGATIVE / ERROR-HANDLING TESTS
5. REGRESSION RISKS — existing functionality the diff could plausibly break, and what to re-test.
6. AUTOMATION CANDIDATES — which of the above are worth automating, and at what level (unit/API/UI).

Format every test case as:
  TC-<n>: <title>
  Preconditions: ...
  Steps: ...
  Expected result: ...

Ground every test case in the actual diff and acceptance criteria — do not invent
features that are not present. If the diff and the ticket contradict each other,
flag it explicitly. Output plain text (no markdown syntax), since this will be
posted as a Jira comment."""


def generate_test_cases(issue: dict, pr: dict | None, diff: str | None) -> str:
    client = Anthropic()  # reads ANTHROPIC_API_KEY from env

    pr_section = "No linked PR was found for this ticket. Base the test cases on the ticket alone, and note that the PR was unavailable."
    if pr:
        pr_section = (
            f"PR #{pr['number']}: {pr.get('title', '')}\n"
            f"Branch: {pr.get('head', {}).get('ref', '')}\n"
            f"PR description:\n{pr.get('body') or '(empty)'}\n\n"
            f"--- DIFF ---\n{diff}"
        )

    user_prompt = (
        f"JIRA TICKET {issue['key']} ({issue['issue_type']})\n"
        f"Summary: {issue['summary']}\n\n"
        f"Description:\n{issue['description'] or '(empty)'}\n\n"
        f"Acceptance criteria:\n{issue['acceptance_criteria'] or '(none found)'}\n\n"
        f"LINKED PULL REQUEST:\n{pr_section}"
    )

    response = client.messages.create(
        model=MODEL,
        max_tokens=MAX_OUTPUT_TOKENS,
        system=SYSTEM_PROMPT,
        messages=[{"role": "user", "content": user_prompt}],
    )
    return "".join(block.text for block in response.content if block.type == "text")


# ---------- Main ----------
def main() -> int:
    print(f"Fetching Jira issue {ISSUE_KEY}...")
    issue = fetch_jira_issue(ISSUE_KEY)

    print("Looking for a PR link in the ticket's comments...")
    pr = None
    diff = None
    pr_location = find_pr_url_in_comments(ISSUE_KEY)
    if pr_location:
        owner, repo, number = pr_location
        print(f"Found PR link in comments: {owner}/{repo}#{number}")
        pr = fetch_pr(owner, repo, number)
    if pr is None:
        print("No PR link in comments; falling back to searching PRs by issue key...")
        pr = find_linked_pr_by_key(ISSUE_KEY)
        if pr:
            owner, repo = GH_REPO.split("/", 1)
    if pr:
        print(f"Using PR #{pr['number']}: {pr.get('title', '')}")
        diff = fetch_pr_diff(owner, repo, pr["number"])
    else:
        print("No linked PR found; generating from ticket only.")

    print("Generating test cases with Claude...")
    test_cases = generate_test_cases(issue, pr, diff)

    header = "🤖 AI-generated test cases (review before use)\n"
    if pr:
        header += f"Based on ticket {ISSUE_KEY} and PR #{pr['number']} ({pr['html_url']})\n"
    else:
        header += f"Based on ticket {ISSUE_KEY} only — no linked PR found.\n"
    header += "-" * 40 + "\n\n"

    print("Posting comment to Jira...")
    post_jira_comment(ISSUE_KEY, header + test_cases)
    print("Done.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
